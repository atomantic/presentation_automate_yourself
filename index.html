jQuery

var $ = function(selector){
  var query = {
    id: document.getElementById,
    cls: document.getElementByClassName,
    tag: document.getElementByTagName
  }
  var type = 'tag'
  if(selector[0]==='#') type = 'id'; 
  if(selector[0]==='.') type = 'cls'; 
  return query[type](selector);
}
