🤔  Code Libraries are Automation!

Before jQuery

I need an element
var foo = document.getElementById('foo');
var bar = document.getElementById('bar');
remove redundancy
var $ = document.getElementById;
var foo = $('foo');
var bar = $('bar');

jQuery

var $ = function(selector){
  var query = {
    id: document.getElementById,
    cls: document.getElementByClassName,
    tag: document.getElementByTagName
  }
  var type = 'tag'
  if(selector[0]==='#') type = 'id'; 
  if(selector[0]==='.') type = 'cls'; 
  return query[type](selector);
}

🤩  now: document.querySelectorAll() - no more jQuery

Let's Automate!
#!/usr/bin/env bash
git clone git@github.com:$1/$2.git
cd $2
if [ -f "README.md" ]; then
    echo "✅ contains README.md"
else
    echo "🚨 no README.md!"
fi
npm i >/dev/null 2>&1
npm test
if [ $? -eq 0 ]; then
    echo "✅ tests pass"
else
    echo "💀 tests failed"
fi
$ ./test.sh atomantic presentation_automate_yourself
✅ contains README.md
✅ tests pass
